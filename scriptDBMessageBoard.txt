CREATE TABLE IF NOT EXISTS  user (
	usId VARCHAR (50),
	usName VARCHAR (40),
	usLastName VARCHAR (40),
	usPassword VARCHAR (40),
	usAdminFlag BOOLEAN,
	usReadOnly BOOLEAN,
	usProfile VARCHAR (10),
	CONSTRAINT pkUser PRIMARY KEY (usId)
);

CREATE TABLE IF NOT EXISTS categorie (
	caId INT(11) NOT NULL AUTO_INCREMENT,
	caName VARCHAR (50),
	usId VARCHAR (50),
	CONSTRAINT pkCategorie PRIMARY KEY (caId),
	CONSTRAINT caFK1 FOREIGN KEY (usId) REFERENCES user(usId) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS  thread (
	thId INT(11) NOT NULL AUTO_INCREMENT,
	thName VARCHAR (50),
	caId INT,
	usId VARCHAR (50),
	CONSTRAINT pkThread PRIMARY KEY (thId),
	CONSTRAINT thFK1 FOREIGN KEY (usId) REFERENCES user(usId) ON UPDATE CASCADE ON DELETE CASCADE,
	CONSTRAINT thFK2 FOREIGN KEY (caId) REFERENCES categorie(caId) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS  post (
	poId INT(11) NOT NULL AUTO_INCREMENT,
	poComment VARCHAR (757),
	poDate DATE,
 	usId VARCHAR (50),
	thId INT,
	CONSTRAINT pkPost PRIMARY KEY (poId),
	CONSTRAINT poFK1 FOREIGN KEY (usId) REFERENCES user(usId) ON UPDATE CASCADE ON DELETE CASCADE,
	CONSTRAINT poFK2 FOREIGN KEY (thId) REFERENCES thread(thId) ON UPDATE CASCADE ON DELETE CASCADE
);
